%

*P2: verify if a number is prime.

let int n,i,m=0,ok=0;
    
write("Enter the number:");
read(n);
m=n/2; 
for i=2 as long as i<=m with i=i+1 do 
{   
	if n%i==0 then{
		write("Number is not prime") ;
		ok=1   ;
		break;
	}
    
   
}
if ok==0 then  
{  
	write("Number is prime");
}
